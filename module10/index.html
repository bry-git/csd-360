this page will allow you to change the CSS properties of certain elements of the page with the JavaScript below. Click the button to toggle the appearance of the page between light and dark mode.

            
        const handleModeSwitch = () => {
            let mode = document.getElementById("toggle-mode")
            if(mode.className == "light") {
                mode.innerText = "Light Mode";
                mode.className = "dark";
                document.body.style.backgroundColor = "#141D26";
                document.body.style.color = "#fff";
            } else {
                mode.innerText = "Dark Mode"
                mode.className = "light";
                document.body.style.backgroundColor = "#fff";
                document.body.style.color = "#000";
            }
        }
